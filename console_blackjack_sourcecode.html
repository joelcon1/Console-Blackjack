#include <iostream>
#include <vector>
#include <ctime>
#include <cstdlib>

using namespace std;

class Player {
    public:
    vector<int> hand;

    void ReceiveCard() {
        int CardValue = (rand() % 13) + 1; // generates a card value between 1 and 13, instead of 0 and 12 (hence the + 1)
        int AceValue = 0;
        int UserInput;
        if (CardValue > 10) { // Jack, Queen, and King are all face cards with value 10
            CardValue = 10;
        }
        else if (CardValue == 1) {
            cout << "You drew an ace! Would you like to count it as 1 or 11? (enter 1 or 11)" << endl;
            while (AceValue == 0) {
                cin >> UserInput;
                if (UserInput == 1 or UserInput == 11) {
                    AceValue = UserInput;
                    CardValue = AceValue;
                }
                else {
                    cout << "You entered an invalid input. Please try again (1 or 11)." << endl;
                }
            }
        }
        hand.push_back(CardValue);
    }
    int TotalValue() {
        int TotalValue = 0;
        for (int CardValue : hand) { // range based for loop (for each card in vector hand)...
            TotalValue += CardValue;
        }
        return TotalValue;
    }
    void ShowHand() {
        cout << "Cards received: ";
        for (int card : hand) {
            cout << card << ", ";
        }
        cout << "(Total: " << TotalValue() << ")" << endl;
    }


};

class Dealer {
public:
    vector<int> hand;
    void ReceiveCard() {
        int CardValue = (rand() % 13) + 1;
        if (CardValue > 10) {
            CardValue = 10;
        }
        else if (CardValue == 1) {
            CardValue = 11;
        }
        hand.push_back(CardValue);
    }



int TotalValue() {
    int TotalValue = 0;
    for (int CardValue : hand) { // range based for loop (for each card in vector hand)...
        TotalValue += CardValue;
    }
    return TotalValue;
}

void ShowHand() {
    cout << "Dealer's hand: ";
    for (int CardValue : hand) {
        cout << CardValue << ", ";
    }
    cout << "(Total: " << TotalValue() << ")" << endl;
}
};

int main() {
    cout << "Welcome to Blackjack!" << endl;
    srand(time(0)); // using srand(time) instead of rand() ensures a more "random" result by using seeding
    cout << "Click Enter to draw cards:" << endl;
    cin.ignore();

    Player user; // object user in class Player
    Dealer computer;
// player and dealer each receive two cards
    user.ReceiveCard();
    computer.ReceiveCard();
    user.ReceiveCard();
    computer.ReceiveCard();
    user.ShowHand();

    cout << "Dealer's hand: " << computer.hand[0] << ", *" << endl << endl; // dealer only shows first card

// player's turn
    char UserChoice;
    while (true) {
        cout << "Would you like to hit or stand? (Enter h/s): ";
        cin >> UserChoice;
        if (UserChoice == 'h') {
            user.ReceiveCard();
            user.ShowHand();
            int FinalValue = user.TotalValue();
            if (FinalValue > 21) {
                cout << "Bust! You lose!" << endl;
                return 0; // program ends if player loses
            }
        }
        else break;
    }
// dealer's turn
    cout << "Dealer's turn! " << endl;
    computer.ShowHand();
    while (computer.TotalValue() < 17) { // dealer is forced to "hit" until the value of the cards is 17 or higher, or the dealer busts, whichever comes first.
        computer.ReceiveCard();
        computer.ShowHand();
    }
    int PlayerTotal = user.TotalValue();
    int DealerTotal = computer.TotalValue();
    if (DealerTotal > 21 or PlayerTotal > DealerTotal) {
        cout << endl << "You Win!";
    }
    else if (PlayerTotal == DealerTotal) {
        cout << "It's a tie! (Push)";
    }
    else {
        cout << endl << "You lose!";
    }
    return 0;
}
